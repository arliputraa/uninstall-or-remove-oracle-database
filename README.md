# Uninstall/Remove Oracle Database 21c from CentOS

### File path directory $ORACLE_HOME / deinstall
    echo $ORACLE_HOME
    #directory uninstall
    echo $ORACLE_HOME/deinstall 


### Login User Oracle
    sudo su oracle
